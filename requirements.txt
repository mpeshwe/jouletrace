# ============================================================================
# Installation Instructions
# ============================================================================
# 
# 1. Create virtual environment:
#    python3.11 -m venv venv
#    source venv/bin/activate  # On Linux/Mac
#    # or
#    .\venv\Scripts\activate  # On Windows
#
# 2. Install dependencies:
#    pip install --upgrade pip
#    pip install -r requirements.txt
#
# 3. Install system dependencies (Ubuntu/Debian):
#    sudo apt-get update
#    sudo apt-get install -y linux-tools-generic linux-tools-$(uname -r)
#
# 4. Install Redis (if not using Docker):
#    sudo apt-get install redis-server
#
# 5. Configure perf permissions:
#    sudo sysctl kernel.perf_event_paranoid=0
#    # Or add user to perf_users group
#
# 6. Verify installation:
#    python -c "from jouletrace.energy import create_energy_meter; print(create_energy_meter())"
#
# ============================================================================
# Docker Installation (Recommended for Production)
# ============================================================================
# Use the provided Dockerfile which includes all system dependencies
# docker-compose up -d --build
#
# ============================================================================
# Notes
# ============================================================================
# - Pinned versions for reproducibility
# - Standard library modules (pathlib, logging, etc.) don't need installation
# - Energy measurement requires Linux with RAPL support
# - Redis must be running before starting Celery workers
# - For production, consider using a requirements.lock file
